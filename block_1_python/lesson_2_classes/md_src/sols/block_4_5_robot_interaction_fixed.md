import matplotlib.pyplot as plt
import math

class CollaborativeRobot:
    """–†–æ–±–æ—Ç, –∫–æ—Ç–æ—Ä—ã–π —É–º–µ–µ—Ç –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–æ–≤–∞—Ç—å —Å –¥—Ä—É–≥–∏–º–∏ —Ä–æ–±–æ—Ç–∞–º–∏"""
    
    def __init__(self, name, start_x=0, start_y=0, color="blue"):
        self.name = name
        self.x = start_x
        self.y = start_y
        self.start_x = start_x
        self.start_y = start_y
        self.color = color
        self.steps_made = 0
        self.path = [(start_x, start_y)]
        self.mission_completed = False
        self.meetings_count = 0
        
        print(f"ü§ñ {name} –≥–æ—Ç–æ–≤ –∫ –∫–æ–º–∞–Ω–¥–Ω–æ–π —Ä–∞–±–æ—Ç–µ –≤ ({start_x}, {start_y})")
    
    def get_position(self):
        """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Ç–µ–∫—É—â—É—é –ø–æ–∑–∏—Ü–∏—é"""
        return (self.x, self.y)
    
    def distance_to(self, other_robot):
        """–í—ã—á–∏—Å–ª—è–µ—Ç —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –¥–æ –¥—Ä—É–≥–æ–≥–æ —Ä–æ–±–æ—Ç–∞"""
        dx = self.x - other_robot.x
        dy = self.y - other_robot.y
        return math.sqrt(dx * dx + dy * dy)
    
    def move_towards(self, target_x, target_y):
        """–î–µ–ª–∞–µ—Ç –æ–¥–∏–Ω —à–∞–≥ –∫ —Ü–µ–ª–µ–≤–æ–π —Ç–æ—á–∫–µ"""
        if self.x < target_x:
            self.x += 1
        elif self.x > target_x:
            self.x -= 1
        
        if self.y < target_y:
            self.y += 1
        elif self.y > target_y:
            self.y -= 1
        
        self.steps_made += 1
        self.path.append((self.x, self.y))
        print(f"üö∂ {self.name} –∏–¥–µ—Ç –∫ ({target_x}, {target_y}) ‚Üí —Ç–µ–∫—É—â–∞—è ({self.x}, {self.y})")
    
    def navigate_to_point(self, target_x, target_y, max_steps=50):
        """–ù–∞–≤–∏–≥–∏—Ä—É–µ—Ç –∫ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–π —Ç–æ—á–∫–µ"""
        print(f"üéØ {self.name} –Ω–∞–≤–∏–≥–∏—Ä—É–µ—Ç –∫ ({target_x}, {target_y})")
        
        steps_taken = 0
        
        while (self.x != target_x or self.y != target_y) and steps_taken < max_steps:
            self.move_towards(target_x, target_y)
            steps_taken += 1
        
        if self.x == target_x and self.y == target_y:
            print(f"‚úÖ {self.name} –¥–æ—Å—Ç–∏–≥ —Ü–µ–ª–∏ –∑–∞ {steps_taken} —à–∞–≥–æ–≤!")
            return True
        else:
            print(f"‚è∞ {self.name} –Ω–µ –¥–æ—Å—Ç–∏–≥ —Ü–µ–ª–∏ –∑–∞ {max_steps} —à–∞–≥–æ–≤")
            return False
    
    def show_status(self):

        print(f"üìä {self.name}: –ø–æ–∑–∏—Ü–∏—è ({self.x}, {self.y}), —à–∞–≥–æ–≤: {self.steps_made}, –≤—Å—Ç—Ä–µ—á: {self.meetings_count}")

class RobotManager:
    """–ú–µ–Ω–µ–¥–∂–µ—Ä –¥–ª—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ü–∏–∏ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è —Ä–æ–±–æ—Ç–æ–≤"""
    
    def __init__(self):
        self.robots = []
        self.meetings_coordinated = 0
        self.group_missions = 0
        
        print("üë• –ú–µ–Ω–µ–¥–∂–µ—Ä —Ä–æ–±–æ—Ç–æ–≤ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä–æ–≤–∞–Ω")
    
    def add_robot(self, robot):
        """–î–æ–±–∞–≤–ª—è–µ—Ç —Ä–æ–±–æ—Ç–∞ –ø–æ–¥ —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ"""
        self.robots.append(robot)
        print(f"üìù {robot.name} –¥–æ–±–∞–≤–ª–µ–Ω –≤ –∫–æ–º–∞–Ω–¥—É (–≤—Å–µ–≥–æ —Ä–æ–±–æ—Ç–æ–≤: {len(self.robots)})")
    
    def find_center_point(self, robot1, robot2):
        """–ù–∞—Ö–æ–¥–∏—Ç —Ü–µ–Ω—Ç—Ä–∞–ª—å–Ω—É—é —Ç–æ—á–∫—É –º–µ–∂–¥—É –¥–≤—É–º—è —Ä–æ–±–æ—Ç–∞–º–∏"""
        center_x = (robot1.x + robot2.x) // 2
        center_y = (robot1.y + robot2.y) // 2
        
        print(f"üìê –¶–µ–Ω—Ç—Ä –º–µ–∂–¥—É {robot1.name} –∏ {robot2.name}: ({center_x}, {center_y})")
        return center_x, center_y
    
    def coordinate_meeting(self, robot1, robot2):
        """–ö–æ–æ—Ä–¥–∏–Ω–∏—Ä—É–µ—Ç –≤—Å—Ç—Ä–µ—á—É –¥–≤—É—Ö —Ä–æ–±–æ—Ç–æ–≤"""
        print(f"\nü§ù –ö–û–û–†–î–ò–ù–ê–¶–ò–Ø –í–°–¢–†–ï–ß–ò: {robot1.name} –∏ {robot2.name}")
        
        # –ù–∞—Ö–æ–¥–∏–º —Ç–æ—á–∫—É –≤—Å—Ç—Ä–µ—á–∏
        meeting_x, meeting_y = self.find_center_point(robot1, robot2)
        
        print(f"üìç –¢–æ—á–∫–∞ –≤—Å—Ç—Ä–µ—á–∏: ({meeting_x}, {meeting_y})")
        print(f"üìè –†–∞—Å—Å—Ç–æ—è–Ω–∏–µ –º–µ–∂–¥—É —Ä–æ–±–æ—Ç–∞–º–∏: {robot1.distance_to(robot2):.2f}")
        
        # –û–±–∞ —Ä–æ–±–æ—Ç–∞ –∏–¥—É—Ç –∫ —Ç–æ—á–∫–µ –≤—Å—Ç—Ä–µ—á–∏
        print(f"\nüö∂ {robot1.name} –Ω–∞—á–∏–Ω–∞–µ—Ç –¥–≤–∏–∂–µ–Ω–∏–µ –∫ –≤—Å—Ç—Ä–µ—á–µ...")
        success1 = robot1.navigate_to_point(meeting_x, meeting_y)
        
        print(f"\nüö∂ {robot2.name} –Ω–∞—á–∏–Ω–∞–µ—Ç –¥–≤–∏–∂–µ–Ω–∏–µ –∫ –≤—Å—Ç—Ä–µ—á–µ...")  
        success2 = robot2.navigate_to_point(meeting_x, meeting_y)
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º —É—Å–ø–µ—Ö –≤—Å—Ç—Ä–µ—á–∏
        if success1 and success2:
            robot1.meetings_count += 1
            robot2.meetings_count += 1
            self.meetings_coordinated += 1
            print(f"üéâ –í–°–¢–†–ï–ß–ê –£–°–ü–ï–®–ù–ê! {robot1.name} –∏ {robot2.name} –≤—Å—Ç—Ä–µ—Ç–∏–ª–∏—Å—å –≤ ({meeting_x}, {meeting_y})")
            return True
        else:
            print(f"‚ùå –í—Å—Ç—Ä–µ—á–∞ –Ω–µ —Å–æ—Å—Ç–æ—è–ª–∞—Å—å")
            return False
    
    def coordinate_group_meeting(self, robot_indices=None):
        """–ö–æ–æ—Ä–¥–∏–Ω–∏—Ä—É–µ—Ç –≤—Å—Ç—Ä–µ—á—É –≥—Ä—É–ø–ø—ã —Ä–æ–±–æ—Ç–æ–≤"""
        if robot_indices is None:
            selected_robots = self.robots
        else:
            selected_robots = [self.robots[i] for i in robot_indices if i < len(self.robots)]
        
        if len(selected_robots) < 2:
            print("‚ùå –ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Ä–æ–±–æ—Ç–æ–≤ –¥–ª—è –≥—Ä—É–ø–ø–æ–≤–æ–π –≤—Å—Ç—Ä–µ—á–∏")
            return False
        
        print(f"\nüë• –ì–†–£–ü–ü–û–í–ê–Ø –í–°–¢–†–ï–ß–ê: {[robot.name for robot in selected_robots]}")
        
        # –í—ã—á–∏—Å–ª—è–µ–º —Ü–µ–Ω—Ç—Ä –≥—Ä—É–ø–ø—ã —Ä–æ–±–æ—Ç–æ–≤
        total_x = sum(robot.x for robot in selected_robots)
        total_y = sum(robot.y for robot in selected_robots)
        
        center_x = total_x // len(selected_robots)
        center_y = total_y // len(selected_robots)
        
        print(f"üìç –¶–µ–Ω—Ç—Ä –≥—Ä—É–ø–ø—ã: ({center_x}, {center_y})")
        
        # –í—Å–µ —Ä–æ–±–æ—Ç—ã –∏–¥—É—Ç –∫ —Ü–µ–Ω—Ç—Ä—É
        all_arrived = True
        for robot in selected_robots:
            print(f"\nüö∂ {robot.name} –¥–≤–∏–∂–µ—Ç—Å—è –∫ –≥—Ä—É–ø–ø–æ–≤–æ–º—É —Ü–µ–Ω—Ç—Ä—É...")
            success = robot.navigate_to_point(center_x, center_y)
            if success:
                robot.meetings_count += 1
            else:
                all_arrived = False
        
        if all_arrived:
            self.group_missions += 1
            print(f"üéâ –ì–†–£–ü–ü–û–í–ê–Ø –í–°–¢–†–ï–ß–ê –£–°–ü–ï–®–ù–ê! –í—Å–µ —Ä–æ–±–æ—Ç—ã –≤ —Ü–µ–Ω—Ç—Ä–µ ({center_x}, {center_y})")
            return True
        else:
            print(f"‚ùå –ù–µ –≤—Å–µ —Ä–æ–±–æ—Ç—ã —Å–º–æ–≥–ª–∏ –¥–æ–±—Ä–∞—Ç—å—Å—è –¥–æ —Ü–µ–Ω—Ç—Ä–∞")
            return False

        if len(self.robots) < 2:
            print("‚ùå –ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Ä–æ–±–æ—Ç–æ–≤ –¥–ª—è —Ü–µ–ø–æ—á–∫–∏")

        base_robot = self.robots[0]

class RelayRobot(CollaborativeRobot):
    """–†–æ–±–æ—Ç –¥–ª—è —ç—Å—Ç–∞—Ñ–µ—Ç—ã"""
    
    def __init__(self, name, start_x, start_y, color):
        super().__init__(name, start_x, start_y, color)
        self.has_baton = False
        self.relay_position = 0
    
    def receive_baton(self):
        """–ü–æ–ª—É—á–∞–µ—Ç —ç—Å—Ç–∞—Ñ–µ—Ç–Ω—É—é –ø–∞–ª–æ—á–∫—É"""
        self.has_baton = True
        print(f"üèÉ {self.name} –ø–æ–ª—É—á–∏–ª —ç—Å—Ç–∞—Ñ–µ—Ç–Ω—É—é –ø–∞–ª–æ—á–∫—É!")
    
    def pass_baton(self, next_robot):
        """–ü–µ—Ä–µ–¥–∞–µ—Ç –ø–∞–ª–æ—á–∫—É —Å–ª–µ–¥—É—é—â–µ–º—É —Ä–æ–±–æ—Ç—É"""
        if self.has_baton:
            self.has_baton = False
            next_robot.receive_baton()
            print(f"ü§ù {self.name} –ø–µ—Ä–µ–¥–∞–ª –ø–∞–ª–æ—á–∫—É ‚Üí {next_robot.name}")
            return True
        return False

```python
    def find_safe_direction(self):
        """–ù–∞—Ö–æ–¥–∏—Ç –±–µ–∑–æ–ø–∞—Å–Ω–æ–µ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –¥–ª—è –¥–≤–∏–∂–µ–Ω–∏—è"""
        safe_directions = []
        
        # TODO: –ü—Ä–æ–≤–µ—Ä—å—Ç–µ –≤—Å–µ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è –∏ –Ω–∞–π–¥–∏—Ç–µ –±–µ–∑–æ–ø–∞—Å–Ω—ã–µ

            if direction == "up":

            elif direction == "down":

            elif direction == "left":

            elif direction == "right":
                test_x += 1
            
            # –ü—Ä–æ–≤–µ—Ä—è–µ–º –±–µ–∑–æ–ø–∞—Å–Ω–æ—Å—Ç—å

    def smart_navigate_to(self, target_x, target_y):
        """–£–º–Ω–∞—è –Ω–∞–≤–∏–≥–∞—Ü–∏—è —Å —É—á–µ—Ç–æ–º –≥—Ä–∞–Ω–∏—Ü"""

            print(f"‚ùå –¶–µ–ª—å ({target_x}, {target_y}) –≤–Ω–µ –≥—Ä–∞–Ω–∏—Ü –º–∏—Ä–∞!")

        while (self.x != target_x or self.y != target_y) and attempts < max_attempts:
            attempts += 1
            
            # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –∫ —Ü–µ–ª–∏

                if self.safe_move("right"):
                    continue
            elif self.x > target_x:
                if self.safe_move("left"):
                    continue
            
            if self.y < target_y:
                if self.safe_move("up"):
                    continue
            elif self.y > target_y:
                if self.safe_move("down"):
                    continue
            
            # –ï—Å–ª–∏ –ø—Ä—è–º–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ –Ω–µ–≤–æ–∑–º–æ–∂–Ω–æ, –∏—â–µ–º –æ–±—Ö–æ–¥–Ω–æ–π –ø—É—Ç—å

            if safe_dirs:
                print(f"üîÑ –û–±—Ö–æ–¥–Ω–æ–π –º–∞–Ω–µ–≤—Ä: {safe_dirs[0]}")
                self.safe_move(safe_dirs[0])
            else:
                print(f"üö´ {self.name} –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω! –ù–µ—Ç –¥–æ—Å—Ç—É–ø–Ω—ã—Ö –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–π.")
                break
        
        if self.x == target_x and self.y == target_y:
            print(f"‚úÖ {self.name} –¥–æ—Å—Ç–∏–≥ —Ü–µ–ª–∏ –∑–∞ {attempts} –ø–æ–ø—ã—Ç–æ–∫!")

        else:
            print(f"‚ùå {self.name} –Ω–µ —Å–º–æ–≥ –¥–æ—Å—Ç–∏—á—å —Ü–µ–ª–∏ –∑–∞ {max_attempts} –ø–æ–ø—ã—Ç–æ–∫")
            return False
```

```python
# –ë–ª–æ–∫ 4.4: –ì—Ä–∞–Ω–∏—Ü—ã –º–∏—Ä–∞ - —Ä–æ–±–æ—Ç—ã —É—á–∞—Ç—Å—è —É–≤–∞–∂–∞—Ç—å –ø—Ä–µ–¥–µ–ª—ã! - –†–ï–®–ï–ù–ò–ï
# –°–æ–∑–¥–∞–µ–º –±–µ–∑–æ–ø–∞—Å–Ω—É—é —Ä–∞–±–æ—á—É—é –æ–±–ª–∞—Å—Ç—å –¥–ª—è —Ä–æ–±–æ—Ç–æ–≤

import matplotlib.pyplot as plt
import matplotlib.patches as patches

class BoundedRobot:
    """–†–æ–±–æ—Ç, –∫–æ—Ç–æ—Ä—ã–π –∑–Ω–∞–µ—Ç –≥—Ä–∞–Ω–∏—Ü—ã —Å–≤–æ–µ–≥–æ –º–∏—Ä–∞"""
    
    def __init__(self, name, start_x=0, start_y=0, world_width=10, world_height=8, color="blue"):
        self.name = name
        self.x = start_x
        self.y = start_y
        self.start_x = start_x
        self.start_y = start_y
        self.color = color
        
        # –ì—Ä–∞–Ω–∏—Ü—ã –º–∏—Ä–∞
        self.world_width = world_width
        self.world_height = world_height
        self.min_x = 0
        self.min_y = 0
        self.max_x = world_width - 1
        self.max_y = world_height - 1
        
        # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
        self.steps_made = 0
        self.boundary_violations = 0
        self.path = [(start_x, start_y)]
        self.violation_attempts = []
        
        print(f"üåç {name} —Å–æ–∑–¥–∞–Ω –≤ –º–∏—Ä–µ {world_width}x{world_height}, –ø–æ–∑–∏—Ü–∏—è ({start_x}, {start_y})")
        
        # –ü—Ä–æ–≤–µ—Ä–∏–º, —á—Ç–æ —Å—Ç–∞—Ä—Ç–æ–≤–∞—è –ø–æ–∑–∏—Ü–∏—è –∫–æ—Ä—Ä–µ–∫—Ç–Ω–∞
        if not self.is_position_valid(start_x, start_y):
            print(f"‚ö†Ô∏è –í–ù–ò–ú–ê–ù–ò–ï: {name} —Å–æ–∑–¥–∞–Ω –≤–Ω–µ –≥—Ä–∞–Ω–∏—Ü –º–∏—Ä–∞!")
    
    def is_position_valid(self, x, y):
        """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç, –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –ª–∏ –ø–æ–∑–∏—Ü–∏—è –≤ –≥—Ä–∞–Ω–∏—Ü–∞—Ö –º–∏—Ä–∞"""
        return (self.min_x <= x <= self.max_x) and (self.min_y <= y <= self.max_y)
    
    def get_world_info(self):
        """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ –º–∏—Ä–µ"""
        return {
            'width': self.world_width,
            'height': self.world_height,
            'min_coords': (self.min_x, self.min_y),
            'max_coords': (self.max_x, self.max_y),

    def safe_move(self, direction):

        if direction == "up":
            new_y = self.y + 1
        elif direction == "down":
            new_y = self.y - 1
        elif direction == "left":
            new_x = self.x - 1
        elif direction == "right":
            new_x = self.x + 1
        else:
            print(f"‚ùì {self.name}: –ù–µ–∏–∑–≤–µ—Å—Ç–Ω–æ–µ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ '{direction}'")
            return False
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –º–æ–∂–Ω–æ –ª–∏ –¥–≤–∏–≥–∞—Ç—å—Å—è –≤ –Ω–æ–≤—É—é –ø–æ–∑–∏—Ü–∏—é
        if self.is_position_valid(new_x, new_y):
            # –î–≤–∏–∂–µ–Ω–∏–µ —Ä–∞–∑—Ä–µ—à–µ–Ω–æ
            self.x = new_x
            self.y = new_y
            self.steps_made += 1
            self.path.append((self.x, self.y))
            print(f"‚úÖ {self.name} {direction} ‚Üí ({self.x}, {self.y})")

        else:
            # –î–≤–∏–∂–µ–Ω–∏–µ –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω–æ
            self.boundary_violations += 1
            self.violation_attempts.append({
                'attempted_position': (new_x, new_y),
                'direction': direction,
                'current_position': (self.x, self.y)
            })
            print(f"üö´ {self.name}: –ù–µ–ª—å–∑—è –∏–¥—Ç–∏ {direction}! –ì—Ä–∞–Ω–∏—Ü–∞ –º–∏—Ä–∞ –≤ ({new_x}, {new_y})")
            return False
    
    def explore_boundaries(self):
        """–ò—Å—Å–ª–µ–¥—É–µ—Ç –≥—Ä–∞–Ω–∏—Ü—ã –º–∏—Ä–∞ –≤–æ –≤—Å–µ—Ö –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è—Ö"""

    def walk_perimeter(self):
        """–û–±—Ö–æ–¥–∏—Ç –ø–æ –ø–µ—Ä–∏–º–µ—Ç—Ä—É –º–∏—Ä–∞ (–µ—Å–ª–∏ –≤–æ–∑–º–æ–∂–Ω–æ)"""
        print(f"\nüîÑ {self.name} –ø—ã—Ç–∞–µ—Ç—Å—è –æ–±–æ–π—Ç–∏ –ø–æ –ø–µ—Ä–∏–º–µ—Ç—Ä—É...")
        
        # –ò–¥–µ–º –≤ –ª–µ–≤—ã–π –Ω–∏–∂–Ω–∏–π —É–≥–æ–ª
        while self.x > self.min_x:
            if not self.safe_move("left"):
                break
        while self.y > self.min_y:
            if not self.safe_move("down"):
                break
        
        print(f"–°—Ç–∞—Ä—Ç–æ–≤–∞—è –ø–æ–∑–∏—Ü–∏—è –¥–ª—è –ø–µ—Ä–∏–º–µ—Ç—Ä–∞: ({self.x}, {self.y})")
        
        # 1. –ò–¥–µ–º –≤–ø—Ä–∞–≤–æ –¥–æ –≥—Ä–∞–Ω–∏—Ü—ã
        while self.x < self.max_x:
            if not self.safe_move("right"):
                break
        
        # 2. –ò–¥–µ–º –≤–≤–µ—Ä—Ö –¥–æ –≥—Ä–∞–Ω–∏—Ü—ã
        while self.y < self.max_y:

                break
        
        # 3. –ò–¥–µ–º –≤–ª–µ–≤–æ –¥–æ –≥—Ä–∞–Ω–∏—Ü—ã
        while self.x > self.min_x:

                break
        
        # 4. –ò–¥–µ–º –≤–Ω–∏–∑ –¥–æ –≥—Ä–∞–Ω–∏—Ü—ã
        while self.y > self.min_y:
            if not self.safe_move("down"):
                break
        
        print(f"‚úÖ –û–±—Ö–æ–¥ –ø–µ—Ä–∏–º–µ—Ç—Ä–∞ –∑–∞–≤–µ—Ä—à–µ–Ω!")

        safe_directions = []
        
        directions = ["up", "down", "left", "right"]
        
        for direction in directions:
            # –í—ã—á–∏—Å–ª—è–µ–º –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω—É—é –Ω–æ–≤—É—é –ø–æ–∑–∏—Ü–∏—é
            test_x, test_y = self.x, self.y
            
            if direction == "up":
                test_y += 1
            elif direction == "down":
                test_y -= 1
            elif direction == "left":
                test_x -= 1
            elif direction == "right":
                test_x += 1
            
            # –ü—Ä–æ–≤–µ—Ä—è–µ–º –±–µ–∑–æ–ø–∞—Å–Ω–æ—Å—Ç—å
            if self.is_position_valid(test_x, test_y):
                safe_directions.append(direction)
        
        return safe_directions

        """–£–º–Ω–∞—è –Ω–∞–≤–∏–≥–∞—Ü–∏—è —Å —É—á–µ—Ç–æ–º –≥—Ä–∞–Ω–∏—Ü"""
        print(f"\nüéØ {self.name} –Ω–∞–≤–∏–≥–∏—Ä—É–µ—Ç –∫ ({target_x}, {target_y})")
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ —Ü–µ–ª—å –≤ –ø—Ä–µ–¥–µ–ª–∞—Ö –º–∏—Ä–∞
        if not self.is_position_valid(target_x, target_y):
            print(f"‚ùå –¶–µ–ª—å ({target_x}, {target_y}) –≤–Ω–µ –≥—Ä–∞–Ω–∏—Ü –º–∏—Ä–∞!")
            return False
        
        max_attempts = 100  # –∑–∞—â–∏—Ç–∞ –æ—Ç –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ–≥–æ —Ü–∏–∫–ª–∞
        attempts = 0
        
        while (self.x != target_x or self.y != target_y) and attempts < max_attempts:
            attempts += 1
            
            # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –∫ —Ü–µ–ª–∏
            if self.x < target_x:
                if self.safe_move("right"):
                    continue
            elif self.x > target_x:
                if self.safe_move("left"):
                    continue
            
            if self.y < target_y:
                if self.safe_move("up"):
                    continue
            elif self.y > target_y:
                if self.safe_move("down"):
                    continue
            
            # –ï—Å–ª–∏ –ø—Ä—è–º–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ –Ω–µ–≤–æ–∑–º–æ–∂–Ω–æ, –∏—â–µ–º –æ–±—Ö–æ–¥–Ω–æ–π –ø—É—Ç—å
            safe_dirs = self.find_safe_direction()
            if safe_dirs:
                print(f"üîÑ –û–±—Ö–æ–¥–Ω–æ–π –º–∞–Ω–µ–≤—Ä: {safe_dirs[0]}")
                self.safe_move(safe_dirs[0])
            else:
                print(f"üö´ {self.name} –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω! –ù–µ—Ç –¥–æ—Å—Ç—É–ø–Ω—ã—Ö –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–π.")
                break
        
        if self.x == target_x and self.y == target_y:
            print(f"‚úÖ {self.name} –¥–æ—Å—Ç–∏–≥ —Ü–µ–ª–∏ –∑–∞ {attempts} –ø–æ–ø—ã—Ç–æ–∫!")
            return True
        else:
            print(f"‚ùå {self.name} –Ω–µ —Å–º–æ–≥ –¥–æ—Å—Ç–∏—á—å —Ü–µ–ª–∏ –∑–∞ {max_attempts} –ø–æ–ø—ã—Ç–æ–∫")
            return False
```
